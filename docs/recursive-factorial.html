Java

static int recursiveFactorial(int n) {
    if (n == 1) return 1;
    return n * recursiveFactorial(n - 1);
}

Go

func recursiveFactorial(n int) int {
	if n == 1 {
		return 1
	}
	return n * recursiveFactorial(n-1)
}

Python

def recursive_factorial(n):
